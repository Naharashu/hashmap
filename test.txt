#include "map.h"


#ifndef TABLE_SIZE
#define TABLE_SIZE 131071
#endif

/*
   Hashmap by Naharashu
   Version 1.1
   MIT LICENSE
*/


Entry* table[TABLE_SIZE] = {0};


int hash(char *str) {
	int i = 0;
	int len = strlen(str);
	for(int j = 0; j < len; j++) {
		i = str[j] + (i << 6) + (i << 16) - i;
	}
	return i;
}

int indx(char *str) {
	int a = (unsigned int)(hash(str)) % TABLE_SIZE;
	return a;
}

void set(char *varname, int val) {
	int i = indx(varname);
	Entry* a = table[i];
	if(strcmp(a->key, "") != 0) {
		a->value = val;
		return;
	}
	Entry* e = malloc(sizeof(Entry));
	e->key = strdup(varname);
	e->value = val;
	e->next = table[i];
	table[i] = e;
}

int get(char *name) {
	int i = indx(name);
	Entry* e = table[i];
	while(e) {
		if(strcmp(e->key, name) == 0) {
			return e->value;
		}
		e = e->next;
	}
	return -1;
}

void free_table() {
	for(int i = 0; i < TABLE_SIZE; i++) {
		Entry* e = table[i];
		while(e) {
			Entry* next = e->next;
			free(e->key);
			free(e);
			e = next;
		}
		table[i] = NULL;
	}
}
